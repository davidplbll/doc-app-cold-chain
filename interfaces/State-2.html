import { createSelector, createFeatureSelector } from '@ngrx/store';
import * as deviceListReducer from './device-list.reducer';
import * as deviceReducer from './device.reducer';
import * as metricicListReducer from './metric-list.reducer';
export interface DeviceStateModule {
  deviceList: deviceListReducer.State;
  device: deviceReducer.State;
  metricList: metricicListReducer.State;
}

export interface State {
  DeviceStateModule: DeviceStateModule;
}
export const reducer = {
  deviceList: deviceListReducer.reducer,
  device: deviceReducer.reducer,
  metricList: metricicListReducer.reducer,
};

export const DeviceModuleSelector = createFeatureSelector<DeviceStateModule>(
  'DeviceModule'
);

export const DeviceSelector = createSelector(
  DeviceModuleSelector,
  (state: DeviceStateModule) => state.device
);

export const getDevice = createSelector(
  DeviceSelector,
  deviceReducer.getDevice
);
export const getDeviceLoading = createSelector(
  DeviceSelector,
  deviceReducer.getDeviceLoading
);
export const getDeviceError = createSelector(
  DeviceSelector,
  deviceReducer.getDeviceError
);

export const DeviceListSelector = createSelector(
  DeviceModuleSelector,
  (state: DeviceStateModule) => state.deviceList
);

export const getDeviceList = createSelector(
  DeviceListSelector,
  deviceListReducer.getDeviceList
);
export const getDeviceListLoading = createSelector(
  DeviceListSelector,
  deviceListReducer.getDeviceListLoading
);
export const getDeviceListError = createSelector(
  DeviceListSelector,
  deviceListReducer.getDeviceListError
);

///

export const MetricListSelector = createSelector(
  DeviceModuleSelector,
  (state: DeviceStateModule) => state.metricList
);

export const getMetricList = createSelector(
  MetricListSelector,
  metricicListReducer.getList
);
export const getMetricListLoading = createSelector(
  MetricListSelector,
  metricicListReducer.getLoading
);
export const getMetricListError = createSelector(
  MetricListSelector,
  metricicListReducer.getError
);
