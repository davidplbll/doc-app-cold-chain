import { createReducer, on } from '@ngrx/store';
import { Refrigerator } from '@cold-chain/interfaces';
import {
  loadRefrigerator,
  loadRefrigeratorFailure,
  loadRefrigeratorSuccess,
  createRefrigerator,
  createRefrigeratorFailure,
  createRefrigeratorSuccess,
  updateRefrigerator,
  updateRefrigeratorFailure,
  updateRefrigeratorSuccess,
} from '../actions/refrigerator.actions';
export const refrigeratorFeatureKey = 'refrigerator';

export interface State {
  loading: boolean;
  error: any;
  refrigerator: Refrigerator | null;
}

export const initialState: State = {
  error: null,
  loading: false,
  refrigerator: null,
};

export const reducer = createReducer(
  initialState,

  on(
    loadRefrigerator,
    (state): State => ({
      ...state,
      loading: true,
      refrigerator: null,
    })
  ),
  on(
    loadRefrigeratorSuccess,
    (state, { data }): State => ({
      ...state,
      loading: false,
      refrigerator: data,
    })
  ),
  on(
    loadRefrigeratorFailure,
    (state, { error }): State => ({
      ...state,
      loading: false,
      error,
    })
  ),
  on(
    createRefrigerator,
    (state, { data }): State => ({
      ...state,
      loading: true,
      refrigerator: data,
    })
  ),
  on(
    createRefrigeratorSuccess,
    (state, { data }): State => ({
      ...state,
      loading: false,
      refrigerator: data,
    })
  ),
  on(
    createRefrigeratorFailure,
    (state, { error }): State => ({
      ...state,
      loading: false,
      error,
    })
  ),
  on(
    updateRefrigerator,
    (state, { data }): State => ({
      ...state,
      loading: true,
      refrigerator: data,
    })
  ),
  on(
    updateRefrigeratorFailure,
    (state, { error }): State => ({
      ...state,
      loading: false,
      error,
    })
  ),
  on(
    updateRefrigeratorSuccess,
    (state, { data }): State => ({
      ...state,
      loading: false,
      refrigerator: data,
    })
  )
);

export const getLoading = (state: State) => state.loading;
export const getError = (state: State) => state.error;
export const getRefrigerator = (state: State) => state.refrigerator;
