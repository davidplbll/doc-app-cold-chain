import { SelectionModel } from '@angular/cdk/collections';
import {
  Component,
  EventEmitter,
  Input,
  OnInit,
  Output,
  ViewChild,
  OnChanges,
  SimpleChanges,
} from '@angular/core';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';
import { MatTableDataSource } from '@angular/material/table';

@Component({
  selector: 'cold-chain-dynamic-table',
  templateUrl: './dynamic-table.component.html',
  styleUrls: ['./dynamic-table.component.scss'],
})
export class DynamicTableComponent implements OnChanges {
  @Input() data: any;
  @Input() cols: any[] = [];
  @Input() config: any = null;
  @Input()
  set filter(value: string) {
    if (this.dataSource) {
      this.dataSource.filter = value.trim().toLowerCase();
      if (this.dataSource.paginator) {
        this.dataSource.paginator.firstPage();
      }
    }
  }
  @Input() isLoadingResults: any = false;
  @Output() delete = new EventEmitter<any>();
  @Output() changePage = new EventEmitter<any>();
  @Output() selectItem = new EventEmitter<any>();
  @Output() checkedItems = new EventEmitter<any[]>();
  @Output() openTracking = new EventEmitter<any>();

  @ViewChild(MatPaginator, { static: true }) paginator:
    | MatPaginator
    | undefined = undefined;
  @ViewChild(MatSort, { static: true }) sort: MatSort | undefined = undefined;
  dataSource: MatTableDataSource<any> | any = undefined;
  displayedColumns: any = [];

  selection = new SelectionModel<any>(true, []);

  ngOnChanges(): void {
    this.selection.clear();
    this.displayedColumns = Array.from(
      this.cols ?? [],
      (col) => col.field
    ).concat(this.config?.ocultActions ? [] : ['acciones']);
    if (this.config?.viewAnidated) {
      this.displayedColumns = ['anidated', ...this.displayedColumns];
    }
    if (this.config?.multipleSelect) {
      this.displayedColumns = ['select', ...this.displayedColumns];
    }

    console.log('this.data ', this.data);
    if (this.config?.dinamic) {
      this.dataSource = new MatTableDataSource(this.data?.results ?? []);
    } else {
      this.dataSource = new MatTableDataSource(this.data ?? []);
      console.log('dataSource ', this.data ?? []);
      this.dataSource.paginator = this.paginator;
      console.log('this.dataSource.data ', this.dataSource.data);
    }
    this.dataSource.sort = this.sort;

    console.log('displayedColumns ', this.displayedColumns);
  }
}
