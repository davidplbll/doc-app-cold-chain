import {
  Component,
  Input,
  OnInit,
  OnChanges,
  SimpleChanges,
} from '@angular/core';
import { Store } from '@ngrx/store';
import { loadVaccines, getVaccineList } from '@cold-chain/vaccine-config';
import { Vaccine, VaccineTable } from '@cold-chain/interfaces';
import { AlertService } from '@cold-chain/utils';
@Component({
  selector: 'cold-chain-vaccine-table',
  templateUrl: './vaccine-table.component.html',
  styleUrls: ['./vaccine-table.component.css'],
})
export class VaccineTableComponent implements OnInit, OnChanges {
  @Input() vaccinesInput!: VaccineTable[] | null | undefined;
  vaccines$ = this.store.select(getVaccineList);
  dataSource: VaccineTable[] = [];
  vaccines: Vaccine[] = [];
  cols: { field: string; header: string }[] = [
    { field: 'name', header: 'Nombre' },
    { field: 'amount', header: 'Cantidad' },
    { field: 'maxTemperature', header: 'Temperatura Máxima' },
    { field: 'minTemperature', header: 'Temperatura Mínima' },
  ];
  displayedColumns = [...this.cols.map((col) => col.field), 'delete'];
  constructor(private store: Store, private alertService: AlertService) {}
  ngOnChanges(): void {
    this.dataSource = [];
    this.chargeVaccinesInput();
  }

  ngOnInit(): void {
    this.store.dispatch(loadVaccines());
    this.vaccines$.subscribe((vaccines) => {
      this.vaccines = vaccines;
      this.chargeVaccinesInput();
    });
  }

  async addVaccine() {
    console.log('add');
    const vaccine = await this.alertService.MessageDropdownQuestion(
      'Seleccione Vacuna',
      this.vaccines.map((vaccine) => ({
        label: vaccine?.name ?? '',
        value: vaccine?.id ?? '',
      })),
      'Siguiente'
    );
    if (vaccine) {
      const amount = await this.alertService.MessageNumberQuestion(
        'Ingrese cantidad de vacunas'
      );

      console.log('vaccine ', vaccine);
      console.log('amount ', amount);
      if (amount) {
        const vaccineFind = this.vaccines.find(
          (vaccineFind) => vaccineFind.id === vaccine
        );
        this.dataSource = [
          ...this.dataSource,
          {
            name: vaccineFind?.name ?? '',
            amount,
            maxTemperature: vaccineFind?.config?.maxTemperature,
            minTemperature: vaccineFind?.config?.minTemperature,
            id: vaccineFind?.id ?? '',
          },
        ];
        console.log('vaccineFind ', vaccineFind);
      }
    }
  }

  deleteVaccine(element: VaccineTable) {
    console.log('delete', element);
    this.dataSource = this.dataSource.filter(
      (vaccine) => vaccine?.id !== element?.id
    );
  }

  chargeVaccinesInput() {
    if (this.vaccines) {
      console.log('this.vaccinesInput ', this.vaccinesInput);
      this.vaccinesInput?.forEach((vaccine) => {
        const vaccineFind = this.vaccines.find(
          (vaccineFind) => vaccineFind.id === vaccine.id
        );
        if (
          vaccineFind &&
          !this.dataSource.find((data) => data.id === vaccine.id)
        ) {
          this.dataSource = [
            ...this.dataSource,
            {
              name: vaccineFind?.name ?? '',
              amount: vaccine?.amount ?? 0,
              maxTemperature: vaccineFind?.config?.maxTemperature,
              minTemperature: vaccineFind?.config?.minTemperature,
              id: vaccineFind?.id ?? '',
            },
          ];
        }
      });
    }
  }
}
