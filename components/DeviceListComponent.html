import { Component, OnInit } from '@angular/core';
import { Store } from '@ngrx/store';
import { getDeviceList, getDeviceListLoading } from '../../reducers';
import {
  loadDevices,
  createDevice,
  deleteDevice,
} from '../../actions/device.actions';
import { Device } from '@cold-chain/interfaces';
import { loadBreadCrumb } from '@cold-chain/ui';
import { AlertService } from '@cold-chain/utils';
import { ActivatedRoute, ParamMap, Router } from '@angular/router';
import { map } from 'rxjs/operators';

@Component({
  selector: 'cold-chain-device-list',
  templateUrl: './device-list.component.html',
  styleUrls: ['./device-list.component.scss'],
})
export class DeviceListComponent implements OnInit {
  constructor(
    private store: Store,
    private route: ActivatedRoute,
    private alertService: AlertService,
    private router: Router
  ) {
    store.dispatch(
      loadBreadCrumb({
        data: {
          title: 'Dispositivos',
          icon: 'settings_remote',
          buttonTitle: 'Crear',
          buttonLink: 'dispositives/create',
        },
      })
    );
  }

  list$ = this.store.select(getDeviceList);
  loading$ = this.store.select(getDeviceListLoading);
  cols = [
    {
      field: 'name',
      label: 'Nombre',
      header: 'Nombre',
    },
    {
      field: 'id',
      header: 'ID',
    },
    {
      field: 'run',
      header: 'Tiempo Activo',
    },
  ];
  ngOnInit(): void {
    this.store.dispatch(loadDevices());
    this.route.paramMap
      .pipe(map((params: ParamMap) => params.get('id') ?? ''))
      .subscribe(async (id: string) => {
        console.log('id ', id);
        if (id === 'create') {
          const response = await this.alertService.MessageTextQuestion(
            'Ingrese nombre del dispositivo a crear'
          );
          if (response) {
            this.store.dispatch(
              createDevice({
                data: {
                  name: response,
                },
              })
            );
          } else {
            this.router.navigate(['dispositives']);
          }
        }
      });
  }

  async deleteDevice(event: Device) {
    const response = await this.alertService.MessageConfirm(
      'Seguro de eliminar el dispositivo',
      'ELIMINAR DISPOSITIVO'
    );
    if (response) {
      this.store.dispatch(deleteDevice({ data: event?.id ?? '' }));
    }
  }

  selectDevice(event: Device) {
    this.router.navigate(['dispositives/detail', event?.id]);
  }
}

<div class="container">
    <cold-chain-dynamic-table
      [data]="list$ | async"
      [cols]="cols"
      [isLoadingResults]="loading$ | async"
      (delete)="deleteDevice($event)"
      (selectItem)="selectDevice($event)"
    ></cold-chain-dynamic-table>
  </div>
