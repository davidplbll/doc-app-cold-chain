import {
  Component,
  Input,
  OnInit,
  ViewChild,
  OnChanges,
  SimpleChanges,
} from '@angular/core';
import { FormBuilder, Validators } from '@angular/forms';
import {
  createRefrigerator,
  updateRefrigerator,
} from '../../actions/refrigerator.actions';
import { Store } from '@ngrx/store';
import { loadDevices, getDeviceList } from '@cold-chain/device';
import { VaccineTableComponent } from '../vaccine-table/vaccine-table.component';
import { Refrigerator } from '@cold-chain/interfaces';
@Component({
  selector: 'cold-chain-refrigerator-form',
  templateUrl: './refrigerator-form.component.html',
  styleUrls: ['./refrigerator-form.component.scss'],
})
export class RefrigeratorFormComponent implements OnInit, OnChanges {
  @ViewChild(VaccineTableComponent) vaccineTable!: VaccineTableComponent;
  @Input() refrigerator!: Refrigerator | undefined | null;
  devideList$ = this.store.select(getDeviceList);
  refrigeratorForm = this.fb.group({
    name: ['', Validators.required],
    location: ['', Validators.required],
    device: ['', Validators.required],
  });
  constructor(private fb: FormBuilder, private store: Store) {}
  ngOnChanges(): void {
    this.refrigeratorForm.reset();
    if (this.refrigerator) {
      console.log('this.refrigerator ', this.refrigerator);
      this.refrigeratorForm.patchValue({
        name: this.refrigerator.name,
        location: this.refrigerator.location,
        device: this.refrigerator.device,
      });
    }
  }

  ngOnInit(): void {
    this.store.dispatch(loadDevices());
  }

  onCreateRefrigerator() {
    const data = this.vaccineTable.dataSource;
    console.log('data ', data);
    if (this.refrigerator) {
      this.store.dispatch(
        updateRefrigerator({
          data: {
            ...this.refrigeratorForm.value,
            id: this.refrigerator.id,
            vaccines: data.map((item) => ({
              amount: item.amount,
              id: item.id,
            })),
          },
        })
      );
    } else {
      this.store.dispatch(
        createRefrigerator({
          data: {
            ...this.refrigeratorForm.value,
            vaccines: data.map((item) => ({
              amount: item.amount,
              id: item.id,
            })),
          },
        })
      );
    }
  }
}
