import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, ParamMap } from '@angular/router';
import { Device } from '@cold-chain/interfaces';
import { loadBreadCrumb } from '@cold-chain/ui';
import { Store } from '@ngrx/store';
import { map } from 'rxjs/operators';
import { loadDevice } from '../../actions/device.actions';
import { LoadMetricLists } from '../../actions/metric-list.actions';
import { ApiclientService } from '@cold-chain/api-client';
import { DeviceService } from './../../services';
import { getDevice, getMetricList } from '../../reducers';
@Component({
  selector: 'cold-chain-device',
  templateUrl: './device.component.html',
  styleUrls: ['./device.component.scss'],
})
export class DeviceComponent implements OnInit {
  device$ = this.store.select(getDevice);
  device: Device | null = null;
  metricList$ = this.store.select(getMetricList);

  constructor(
    private store: Store,
    private route: ActivatedRoute,
    private deviceService: DeviceService
  ) {
    this.store.dispatch(
      loadBreadCrumb({
        data: {
          title: 'Dispositivo',
          icon: 'settings_remote',
        },
      })
    );
    deviceService.subscribenewMetrics().subscribe((data) => {
      console.log('data ', data);
    });
  }

  ngOnInit(): void {
    this.route.paramMap
      .pipe(map((params: ParamMap) => params.get('id') ?? ''))
      .subscribe((id: string) => {
        this.store.dispatch(loadDevice({ data: id }));
        this.store.dispatch(LoadMetricLists({ data: id }));
      });
    this.device$.subscribe((response) => {
      console.log('response ', response);
      this.store.dispatch(
        loadBreadCrumb({
          data: {
            title: response?.name ?? 'Dispositivo',
            icon: 'settings_remote',
          },
        })
      );
    });
  }
}
