import {
  Component,
  Input,
  OnInit,
  OnChanges,
  SimpleChanges,
  ViewChild,
  ElementRef,
} from '@angular/core';
import { format } from 'date-fns';
import { Metric } from '@cold-chain/interfaces';

import * as echarts from 'echarts';
@Component({
  selector: 'cold-chain-metrics-device',
  templateUrl: './metrics-device.component.html',
  styleUrls: ['./metrics-device.component.scss'],
})
export class MetricsDeviceComponent implements OnChanges {
  @Input() metrics: Metric[] = [];
  @ViewChild('temperature')
  temperatureElement!: ElementRef;
  @ViewChild('humidity')
  humidityElement!: ElementRef;
  @ViewChild('indexHeat')
  indexHeatElement!: ElementRef;
  @ViewChild('temperatureHistory')
  temperatureHistoryElement!: ElementRef;

  chartTemperature: any = null;
  chartHumidity: any = null;
  chartIndexHeat: any = null;
  chartTemperatureHistory: any = null;

  temperatureConfig = {
    series: [
      {
        type: 'gauge',
        center: ['50%', '60%'],
        startAngle: 200,
        endAngle: -20,
        min: 0,
        max: 40,
        splitNumber: 10,
        itemStyle: {
          color: '#FFAB91',
        },
        progress: {
          show: true,
          width: 30,
        },

        pointer: {
          show: false,
        },
        axisLine: {
          lineStyle: {
            width: 30,
          },
        },
        axisTick: {
          distance: -45,
          splitNumber: 5,

            width: 2,
            color: '#999',
          },
        },
        splitLine: {
          distance: -52,
          length: 14,
          lineStyle: {
            width: 3,
            color: '#999',
          },
        },
        axisLabel: {
          distance: -20,
          color: '#999',
          fontSize: 20,
        },
        anchor: {
          show: false,
        },
        title: {
          show: false,
        },
        detail: {
          valueAnimation: true,
          width: '70%',
          lineHeight: 40,
          height: '15%',
          borderRadius: 8,
          offsetCenter: [0, '-15%'],
          fontSize: 60,
          fontWeight: 'bolder',
          formatter: '{value} °C',
          color: 'auto',
        },
        data: [
          {
            value: 0,
          },
        ],
      },

      {
        type: 'gauge',
        center: ['50%', '60%'],
        startAngle: 200,
        endAngle: -20,
        min: 0,
        max: 40,
        itemStyle: {
          color: '#FD7347',
        },
        progress: {
          show: true,
          width: 8,
        },

        pointer: {
          show: false,
        },
        axisLine: {
          show: false,
        },
        axisTick: {
          show: false,
        },
        splitLine: {
          show: false,
        },
        axisLabel: {
          show: false,
        },
        detail: {
          show: false,
        },
        data: [
          {
            value: 0,
          },
        ],
      },
    ],
  };
  temperatureHistoryConfig: any = {
    title: {
      text: 'HISTORICO DE TEMPERATURAS',
      left: 10,
    },
    toolbox: {
      feature: {
        dataZoom: {
          yAxisIndex: false,
        },
        saveAsImage: {
          pixelRatio: 2,
        },
      },
    },
    tooltip: {
      trigger: 'axis',
      axisPointer: {
        type: 'shadow',
      },
    },
    grid: {
      bottom: 90,
    },
    dataZoom: [
      {
        type: 'inside',
      },
      {
        type: 'slider',
      },
    ],
    xAxis: {
      data: [],
      silent: false,
      splitLine: {
        show: false,
      },
      splitArea: {
        show: false,
      },
    },
    yAxis: {
      splitArea: {
        show: false,
      },
    },
    series: [
      {
        type: 'bar',
        data: [],
        // Set `large` for large data amount
        large: true,
      },
    ],
  };

  ngOnChanges(): void {
    if (this.metrics) {
      const [actualItem] = this.metrics;
      const data = this.metrics.map((metric) => ({
        date: format(new Date(metric?.createdat * 1000), 'MM/dd/yyyy HH:mm:ss'),
        value: metric?.Temperature,
      }));
      if (actualItem) {
        if (!this.chartTemperature) {
          this.chartTemperature = echarts.init(
            this.temperatureElement.nativeElement
          );
        }
        if (!this.chartHumidity) {
          this.chartHumidity = echarts.init(this.humidityElement.nativeElement);
        }
        if (!this.chartIndexHeat) {
          this.chartIndexHeat = echarts.init(
            this.indexHeatElement.nativeElement
          );
        }

        this.chartTemperature.setOption(
          this.setOptionConfig(
            actualItem?.Temperature,
            40,
            10,
            '{value} °C',
            'TEMPERATURA'
          )
        );
        this.chartHumidity.setOption(
          this.setOptionConfig(
            actualItem?.Humidity,
            100,
            10,
            '{value}%',
            'HUMEDAD'
          )
        );
        this.chartIndexHeat.setOption(
          this.setOptionConfig(
            actualItem?.IndexHeat,
            100,
            10,
            '{value}%',
            'ÍNDICE CALOR'
          )
        );
      }

      this.temperatureHistoryConfig = {
        ...this.temperatureHistoryConfig,
        xAxis: {
          ...this.temperatureHistoryConfig?.xAxis,
          data: data.map((item) => item.date).reverse(),
        },
        series: [
          {
            type: 'bar',
            data: data.map((item) => item.value).reverse(),
            // Set `large` for large data amount
            large: true,
          },
        ],
      };
      if (!this.chartTemperatureHistory) {
        this.chartTemperatureHistory = echarts.init(
          this.temperatureHistoryElement.nativeElement
        );

        console.log(
          'this.chartTemperatureHistory ',
          this.chartTemperatureHistory
        );
      }
      this.chartTemperatureHistory.setOption(this.temperatureHistoryConfig);
    }
  }

  setOptionConfig(
    value: number,
    max: number,
    splitNumber: number,
    formatter: string,
    text: string
  ) {
    const [series1, series2] = this.temperatureConfig?.series;
    return {
      title: {
        text,
        left: 10,
      },
      series: [
        {
          ...series1,
          max,
          splitNumber,
          detail: {
            ...series1?.detail,
            formatter,
          },
          data: [
            {
              value,
            },
          ],
        },
        {
          ...series2,
          max,
          data: [
            {
              value,
            },
          ],
        },
      ],
    };
  }
}
