import { Component, Input, OnChanges } from '@angular/core';
import { FormBuilder, Validators } from '@angular/forms';
import { createVaccine, editVaccine } from '../../actions';
import { Store } from '@ngrx/store';
import { Vaccine } from '@cold-chain/interfaces';

@Component({
  selector: 'cold-chain-vaccine-form',
  templateUrl: './vaccine-form.component.html',
  styleUrls: ['./vaccine-form.component.scss'],
})
export class VaccineFormComponent implements OnChanges {
  @Input() vaccine!: Vaccine | null;
  vaccineForm = this.fb.group({
    name: ['', Validators.required],
    maxTemperature: [null, [Validators.required]],
    minTemperature: [null, [Validators.required]],
  });
  constructor(private fb: FormBuilder, private store: Store) {}
  ngOnChanges(): void {
    this.vaccineForm.reset();
    if (this.vaccine) {
      console.log('this.vaccine ', this.vaccine);
      const { name, config } = this.vaccine;
      console.log('config ', config);
      this.vaccineForm.patchValue({
        name,
        ...config,
      });
    }
  }

  onCreateVaccine() {
    const { name, ...config } = this.vaccineForm.value;
    if (this.vaccine) {
      this.store.dispatch(
        editVaccine({
          data: {
            name,
            config,
            id: this.vaccine.id,
          },
        })
      );
    } else {
      this.store.dispatch(
        createVaccine({
          data: {
            name,
            config,
          },
        })
      );
    }
  }
}
