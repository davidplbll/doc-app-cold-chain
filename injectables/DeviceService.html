import { Injectable } from '@angular/core';
import { ApiclientService } from '@cold-chain/api-client';
import { Device, Metric } from '@cold-chain/interfaces';
import { Observable } from 'rxjs';
import { map, tap } from 'rxjs/operators';
@Injectable({
  providedIn: 'root',
})
export class DeviceService {
  constructor(private apiclientService: ApiclientService) {}

  getDevice(id: string) {
    const query = `
    query Query($deviceId: String!) {
      device(id: $deviceId) {
        id
        name
        run
      }
    }`;

    return this.apiclientService
      .query<{ data: { device: Device } }>(
        query,
        'Error al traer el dispositivo',
        {
          deviceId: id,
        }
      )
      .pipe(
        tap(console.log),
        map((data) => data?.data?.device)
      );
  }
  getDeviceList(): Observable<{ data: Device[] }> {
    const query = `query Query {
      devices {
        id
        name
        run
      }
    }`;

    return this.apiclientService
      .query<{ data: { devices: Device[] } }>(
        query,
        'Error al traer el dispositivo'
      )
      .pipe(
        tap(console.log),
        map((data) => ({
          data: data?.data?.devices,
        }))
      );
  }

  subscribenewMetrics(){
    const query = `subscription Subscription {
      newmetrics {
        id
        Humidity
        IndexHeat
        Temperature
        Run
        createdat
      }
    }`;
    return this.apiclientService.suscription(query, 'Error al subscribir');
  }

  getMetricList(id: string): Observable<{ data: Metric[] }> {
    const query = `query Query($metricsId: String!) {
      metrics(id: $metricsId) {
        id
        Humidity
        IndexHeat
        Temperature
        Run
        createdat
      }
    }`;

    return this.apiclientService
      .query<{ data: { devices: Device[] } }>(
        query,
        'Error al traer el dispositivo',
        { metricsId: id }
      )
      .pipe(
        tap(console.log),
        map((data) => ({
          data: data?.data?.metrics,
        }))
      );
  }

  createDevice(device: Device) {
    const query = `mutation Mutation($createdeviceData: DeviceInput!) {
      createdevice(data: $createdeviceData) {
        id
        name
        run
      }
    }`;
    return this.apiclientService.mutation(
      query,
      {
        createdeviceData: device,
      },
      'Error al crear el dispositivo'
    );
  }

  editDevice(device: Device) {
    const query = `mutation EditdeviceMutation($editdeviceData: DeviceInput) {
      editdevice(data: $editdeviceData) {
        id
        name
        run
      }
    }`;
    return this.apiclientService.mutation(
      query,
      {
        editdeviceData: device,
      },
      'Error al editar el dispositivo'
    );
  }

  deleteDevice(id: string) {
    const query = `mutation DeletedeviceMutation($deletedeviceId: String) {
        deletedevice(id: $deletedeviceId)
      }`;
    return this.apiclientService.mutation(
      query,
      {
        deletedeviceId: id,
      },
      'Error al eliminar el dispositivo'
    );
  }
}
