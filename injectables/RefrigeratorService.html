import { Injectable } from '@angular/core';
import { ApiclientService } from '@cold-chain/api-client';
import { Refrigerator } from '@cold-chain/interfaces';
import { Observable } from 'rxjs';
import { map, tap } from 'rxjs/operators';
@Injectable({
  providedIn: 'root',
})
export class RefrigeratorService {
  constructor(private api: ApiclientService) {}

  getRefrigeratorList(): Observable<Refrigerator[]> {
    return this.api
      .query(
        `{
          refrigerators{
            id
            name
            location
            device
            vaccines{
              id
              amount
            }
          }
        }
    `,
        'Error al consultar'
      )
      .pipe(map((response) => response?.data?.refrigerators));
  }
  getRefrigerator(id: string): Observable<Refrigerator> {
    return this.api
      .query(
        `{
          refrigerator(id:"${id}"){
            id
            name
            location
            device
            vaccines{
              id
              amount
            }
          }
        }
    `,
        'Error al consultar'
      )
      .pipe(map((response) => response?.data?.refrigerator));
  }

  createRefrigerator(data: Refrigerator): Observable<Refrigerator> {
    const mutation = `mutation createRefrigerator($refrigerator:RefrigeratorInput!){
      createRefrigerator(data:$refrigerator){
        id
        name
      }
    }`;
    return this.api.mutation(
      mutation,
      { refrigerator: data },
      'Error al crear el refrigerador'
    );
  }

  editRefrigerator(data: Refrigerator): Observable<Refrigerator> {
    const mutation = `mutation editRefrigerator($refrigerator:RefrigeratorInput!){
      editRefrigerator(data:$refrigerator){
        id
        name
        device
      }
    }`;
    return this.api.mutation(
      mutation,
      { refrigerator: data },
      'Error al crear el refrigerador'
    );
  }

  deleteRefrigerator(id: string): Observable<Refrigerator> {
    const mutation = `mutation {
      deleteRefrigerator(id:"${id}")
    }`;
    return this.api.mutation(mutation, {}, 'Error al crear el refrigerador');
  }
}
