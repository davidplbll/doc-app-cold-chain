import { Injectable } from '@angular/core';
import { Actions, createEffect, ofType } from '@ngrx/effects';
import { RefrigeratorService } from '../services/refrigerator.service';
import {
  createRefrigeratorFailure,
  createRefrigeratorSuccess,
  deleteRefrigeratorFailure,
  deleteRefrigeratorSuccess,
  loadRefrigerators,
  RefrigeratorActions,
  loadRefrigeratorFailure,
  loadRefrigeratorSuccess,
  updateRefrigeratorFailure,
  updateRefrigeratorSuccess,
} from '../actions/refrigerator.actions';
import { Router } from '@angular/router';
import { catchError, map, switchMap, tap } from 'rxjs/operators';
import { AlertService } from '@cold-chain/utils';
import { of } from 'rxjs';

@Injectable()
export class RefrigeratorEffects {
  createRefrigerator$ = createEffect(() => {
    return this.actions$.pipe(
      ofType(RefrigeratorActions.createRefrigerator),
      map((action: any) => action?.data),
      switchMap((refrigerator) =>
        this.refrigeratorService.createRefrigerator(refrigerator).pipe(
          map((data) => createRefrigeratorSuccess({ data })),
          catchError((error) => of(createRefrigeratorFailure({ error })))
        )
      )
    );
  });
  createRefrigeratorSuccess$ = createEffect(
    () => {
      return this.actions$.pipe(
        ofType(RefrigeratorActions.createRefrigeratorSuccess),
        tap(() => {
          this.alertService.Messagesuccess('Refrigerador creado con exito');
          this.router.navigate(['refrigerator']);
        })
      );
    },
    { dispatch: false }
  );
  updateRefrigerator$ = createEffect(() => {
    return this.actions$.pipe(
      ofType(RefrigeratorActions.updateRefrigerator),
      map((action: any) => action?.data),
      switchMap((refrigerator) =>
        this.refrigeratorService.editRefrigerator(refrigerator).pipe(
          map((data) => updateRefrigeratorSuccess({ data })),
          catchError((error) => of(updateRefrigeratorFailure({ error })))
        )
      )
    );
  });
  updateRefrigeratorSuccess$ = createEffect(
    () => {
      return this.actions$.pipe(
        ofType(RefrigeratorActions.updateRefrigeratorSuccess),
        tap(() => {
          this.alertService.Messagesuccess('Refrigerador editado con exito');
          this.router.navigate(['refrigerator']);
        })
      );
    },
    { dispatch: false }
  );

  deleteRefrigerator$ = createEffect(() => {
    return this.actions$.pipe(
      ofType(RefrigeratorActions.deleteRefrigerator),
      map((action: any) => action?.data),
      switchMap((refrigerator) =>
        this.refrigeratorService.deleteRefrigerator(refrigerator?.id).pipe(
          map(() => deleteRefrigeratorSuccess()),
          catchError((error) => of(deleteRefrigeratorFailure({ error })))
        )
      )
    );
  });
  loadRefrigerator$ = createEffect(() => {
    return this.actions$.pipe(
      ofType(RefrigeratorActions.loadRefrigerator),
      map((action: any) => action?.data),
      switchMap((id) =>
        this.refrigeratorService.getRefrigerator(id).pipe(
          map((refrigerator) =>
            loadRefrigeratorSuccess({ data: refrigerator })
          ),
          catchError((error) => of(loadRefrigeratorFailure({ error })))
        )
      )
    );
  });

  deleteRefrigeratorSuccess$ = createEffect(() => {
    return this.actions$.pipe(
      ofType(RefrigeratorActions.deleteRefrigeratorSuccess),
      tap(() => {
        this.alertService.Messagesuccess('Refrigerador eliminado con exito');
      }),
      map(() => loadRefrigerators())
    );
  });

  constructor(
    private actions$: Actions,
    private refrigeratorService: RefrigeratorService,
    private alertService: AlertService,
    private router: Router
  ) {}
}
