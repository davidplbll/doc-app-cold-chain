import { Injectable } from '@angular/core';
import sweetalert2 from 'sweetalert2';

@Injectable({
  providedIn: 'root',
})
export class AlertService {
  MessageError(message: string): void {
    sweetalert2({
      type: 'error',
      title: 'Error',
      confirmButtonText: 'Cerrar',
      text: message,
    }).catch((err) => console.log(err));
  }

  Messagesuccess(message: string): void {
    sweetalert2({
      type: 'success',
      title: 'Completado',
      text: message,
      timer: 2000,
      showConfirmButton: false,
    }).catch((err) => console.log(err));
  }

  MessageConfirm(text: string, title: string) {
    return sweetalert2({
      type: 'warning',
      title,
      text,
      showCancelButton: true,
      confirmButtonText: 'Confirmar',
      cancelButtonText: 'Cancelar',
    }).then((response) => response.value);
  }

  MessageTextQuestion(title: string) {
    return sweetalert2({
      title,
      input: 'text',
      inputAttributes: {
        autocapitalize: 'off',
        type: 'number',
      },
      showCancelButton: true,
      confirmButtonText: 'Crear',
      showLoaderOnConfirm: true,
    }).then((response) => response.value);
  }

  MessageNumberQuestion(title: string) {
    return sweetalert2({
      title,
      input: 'number',
      inputAttributes: {
        autocapitalize: 'off',
        type: 'number',
      },
      inputValidator: (value) => {
        return new Promise((resolve: Function) => {
          if (value) {
            resolve();
          } else {
            resolve('Debes ingresar un numero');
          }
        });
      },
      showCancelButton: true,
      confirmButtonText: 'Crear',
      showLoaderOnConfirm: true,
    }).then((response) => Number(response.value));
  }

  MessageDropdownQuestion(
    title: string,
    options: { label: string; value: any }[],
    confirmButtonText: string
  ) {
    return sweetalert2({
      title,
      input: 'select',
      inputOptions: options.reduce((acc: any, option) => {
        acc[option.value] = option.label;
        return acc;
      }, {}),
      inputAttributes: {
        autocapitalize: 'off',
      },
      showCancelButton: true,
      confirmButtonText,
      showLoaderOnConfirm: true,
      inputValidator: (value) => {
        return new Promise((resolve: Function) => {
          if (value) {
            resolve();
          } else {
            resolve('Debes seleccionar una opci√≥n');
          }
        });
      },
    }).then((response) => response.value);
  }
}
