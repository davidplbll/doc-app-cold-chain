import { Injectable } from '@angular/core';
import { Observable, throwError } from 'rxjs';
import { catchError } from 'rxjs/operators';
import { Apollo } from 'apollo-angular';
import { AlertService } from '@cold-chain/utils';
import gql from 'graphql-tag';
import { Router } from '@angular/router';

@Injectable({
  providedIn: 'root',
})
export class ApiclientService {
  constructor(
    private apollo: Apollo,
    private alertService: AlertService,
    private router: Router
  ) {}

  query<T>(query: string, error: string, data?: any): Observable<any> {
    console.log('query ', query);
    let dataQuery: any = {
      query: gql`
        ${query}
      `,
    };

    console.log('dataQuery ', dataQuery);
    if (data) {
      dataQuery = {
        ...dataQuery,
        variables: {
          ...data,
        },
      };
    }

    console.log('dataQuery ', dataQuery);
    return this.apollo.query<T>(dataQuery).pipe(
      catchError(({ graphQLErrors, networkError }) => {
        console.log('networkError ', networkError);
        console.log('graphQLErrors ', graphQLErrors);
        const loginError = graphQLErrors.find(
          (error: any) => error.message == 'TokenExpiredError'
        );
        if (loginError) {
          this.router.navigateByUrl('/login');
          this.alertService.MessageError(
            'Sesión expirada, vuelva a iniciar sesión'
          );
        } else {
          this.alertService.MessageError(error);
        }
        return throwError(graphQLErrors);
      })
    );
  }

  mutation<T>(mutation: string, data: any, error: string): Observable<any> {
    return this.apollo
      .mutate<T>({
        mutation: gql`
          ${mutation}
        `,
        variables: data,
      })
      .pipe(
        catchError((err) => {
          this.alertService.MessageError(error);
          return throwError(err);
        })
      );
  }

  suscription<T>(query: string, error: string): Observable<any> {
    let dataQuery: any = {
      subscription: gql`
        ${query}
      `,
    };
    return this.apollo.subscribe<T>(dataQuery).pipe(
      catchError((err) => {
        this.alertService.MessageError(error);
        return throwError(err);
      })
    );
  }
}
